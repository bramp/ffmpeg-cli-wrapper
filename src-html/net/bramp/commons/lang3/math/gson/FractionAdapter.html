001package net.bramp.commons.lang3.math.gson;
002
003import com.google.gson.TypeAdapter;
004import com.google.gson.stream.JsonReader;
005import com.google.gson.stream.JsonToken;
006import com.google.gson.stream.JsonWriter;
007import org.apache.commons.lang3.math.Fraction;
008
009import java.io.IOException;
010
011/**
012 * GSON TypeAdapter for Apache Commons Math Fraction Object
013 *
014 * @author bramp
015 *
016 */
017public class FractionAdapter extends TypeAdapter<Fraction> {
018
019  /**
020   * If set, 0/0 returns this value, instead of throwing a ArithmeticException
021   */
022  private final Fraction zeroByZero;
023
024  /**
025   * If set, N/0 returns this value, instead of throwing a ArithmeticException
026   */
027  private final Fraction divideByZero;
028
029  public FractionAdapter() {
030    this(Fraction.ZERO, Fraction.ZERO);
031  }
032
033  private FractionAdapter(Fraction zeroByZero, Fraction divideByZero) {
034    this.zeroByZero = zeroByZero;
035    this.divideByZero = divideByZero;
036  }
037
038  public Fraction read(JsonReader reader) throws IOException {
039    JsonToken next = reader.peek();
040
041    if (next == JsonToken.NULL) {
042      reader.nextNull();
043      return null;
044    }
045
046    if (next == JsonToken.NUMBER) {
047      return Fraction.getFraction(reader.nextDouble());
048    }
049
050    String fraction = reader.nextString().trim();
051
052    // Ambiguous as to what 0/0 is, but FFmpeg seems to think it's zero
053    if (zeroByZero != null && "0/0".equals(fraction)) {
054      return zeroByZero;
055    }
056
057    // Another edge cases is invalid files sometimes output 1/0.
058    if (divideByZero != null && fraction.endsWith("/0")) {
059      return divideByZero;
060    }
061
062    return Fraction.getFraction(fraction);
063  }
064
065  public void write(JsonWriter writer, Fraction value) throws IOException {
066    if (value == null) {
067      writer.nullValue();
068      return;
069    }
070    writer.value(value.toProperString());
071  }
072}
