001package net.bramp.ffmpeg.nut;
002
003
004import com.google.common.base.MoreObjects;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.io.IOException;
009
010public class Packet {
011
012  final static Logger LOG = LoggerFactory.getLogger(Packet.class);
013
014  public enum Startcode {
015    MAIN(0x7A561F5F04ADL + (((long) ('N' << 8) + 'M') << 48)), STREAM(
016        0x11405BF2F9DBL + (((long) ('N' << 8) + 'S') << 48)), SYNCPOINT(
017        0xE4ADEECA4569L + (((long) ('N' << 8) + 'K') << 48)), INDEX(
018        0xDD672F23E64EL + (((long) ('N' << 8) + 'X') << 48)), INFO(
019        0xAB68B596BA78L + (((long) ('N' << 8) + 'I') << 48));
020
021    private final long startcode;
022
023    Startcode(long startcode) {
024      this.startcode = startcode;
025    }
026
027    public long value() {
028      return startcode;
029    }
030
031    public boolean equals(long startcode) {
032      return this.startcode == startcode;
033    }
034
035    /**
036     * Returns the Startcode enum for this code.
037     * 
038     * @param startcode The numeric code for this Startcode.
039     * @return The Startcode
040     */
041    public static Startcode of(long startcode) {
042      for (Startcode c : Startcode.values()) {
043        if (c.equals(startcode)) {
044          return c;
045        }
046      }
047      return null;
048    }
049
050    public static boolean isPossibleStartcode(long startcode) {
051      return (startcode & 0xFFL) == 'N';
052    }
053
054    public static String toString(long startcode) {
055      Startcode c = of(startcode);
056      if (c != null) {
057        return c.name();
058      }
059      return String.format("%X", startcode);
060    }
061  }
062
063  public final PacketHeader header = new PacketHeader();
064  public final PacketFooter footer = new PacketFooter();
065
066
067  protected void readBody(NutDataInputStream in) throws IOException {
068    // Default implementation does nothing
069  }
070
071  public void read(NutDataInputStream in, long startcode) throws IOException {
072    header.read(in, startcode);
073    readBody(in);
074    seekToPacketFooter(in);
075    footer.read(in);
076  }
077
078  public void seekToPacketFooter(NutDataInputStream in) throws IOException {
079    long current = in.offset();
080    if (current > header.end) {
081      throw new IOException("Can not seek backwards at:" + current + " end:" + header.end);
082    }
083    // TODO Fix this to not cast longs to ints
084    in.skipBytes((int) (header.end - current));
085  }
086
087  @Override
088  public String toString() {
089    return MoreObjects.toStringHelper(this).add("header", header).add("footer", footer).toString();
090  }
091}
