001package net.bramp.ffmpeg.nut;
002
003import com.google.common.base.MoreObjects;
004
005public class FrameCode {
006
007  long flags;
008  int streamId;
009  int dataSizeMul;
010  int dataSizeLsb;
011  long ptsDelta;
012  int reservedCount;
013  long matchTimeDelta;
014  int headerIdx;
015
016  @Override
017  public String toString() {
018    return MoreObjects.toStringHelper(this).add("flags", flags).add("id", streamId)
019        .add("dataSizeMul", dataSizeMul).add("dataSizeLsb", dataSizeLsb).add("ptsDelta", ptsDelta)
020        .add("reservedCount", reservedCount).add("matchTimeDelta", matchTimeDelta)
021        .add("headerIdx", headerIdx).toString();
022  }
023}
