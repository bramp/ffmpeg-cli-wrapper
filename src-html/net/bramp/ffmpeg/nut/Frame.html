011/**
012 * A video or audio frame
013 */
014public class Frame {
015  // TODO Change this to a enum
016  static final long FLAG_KEY = 1 << 0;
017  static final long FLAG_EOR = 1 << 1;
018  static final long FLAG_CODED_PTS = 1 << 3;
019  static final long FLAG_STREAM_ID = 1 << 4;
020  static final long FLAG_SIZE_MSB = 1 << 5;
021  static final long FLAG_CHECKSUM = 1 << 6;
022  static final long FLAG_RESERVED = 1 << 7;
023  static final long FLAG_SM_DATA = 1 << 8;
024  static final long FLAG_HEADER_IDX = 1 << 10;
025  static final long FLAG_MATCH_TIME = 1 << 11;
026  static final long FLAG_CODED = 1 << 12;
027  static final long FLAG_INVALID = 1 << 13;
028
029  Stream stream;
030  long flags;
031  long pts;
032  byte[] data;
033
034  Map<String, Object> sideData;
035  Map<String, Object> metaData;
036
037  protected Map<String, Object> readMetaData(NutDataInputStream in) throws IOException {
038    Map<String, Object> data = new TreeMap<String, Object>();
039    long count = in.readVarLong();
040    for (int i = 0; i < count; i++) {
041      byte[] name = in.readVarArray();
042      long type = in.readSignedVarInt();
043      Object value;
044
045      if (type == -1) {
046        value = new String(in.readVarArray(), StandardCharsets.UTF_8);
047
048      } else if (type == -2) {
049        String k = new String(in.readVarArray());
050        String v = new String(in.readVarArray());
051        value = k + "=" + v; // TODO Change this some how
052
053      } else if (type == -3) {
054        value = in.readSignedVarInt();
055
056      } else if (type == -4) {
057        /*
058         * t (v coded universal timestamp) tmp v id= tmp % time_base_count value= (tmp /
059         * time_base_count) * timeBase[id]
060         */
061        value = in.readVarLong(); // TODO Convert to timestamp
062
063      } else if (type < -4) {
064        long denominator = -type - 4;
065        long numerator = in.readSignedVarInt();
066        value = Fraction.getFraction((int) numerator, (int) denominator);
067
068      } else {
069        value = type;
070      }
071
072      data.put(new String(name), value);
073    }
074    return data;
075  }
076
077
078  public void read(NutReader nut, NutDataInputStream in, int code) throws IOException {
079    if (code == 'N') {
080      throw new IOException("Illegal frame code: " + code);
081    }
082
083    FrameCode fc = nut.header.frameCodes.get(code);
084    flags = fc.flags;
085    if ((flags & FLAG_INVALID) == FLAG_INVALID) {
086      throw new IOException("Using invalid framecode: " + code);
087    }
088
089    if ((flags & FLAG_CODED) == FLAG_CODED) {
090      long coded_flags = in.readVarLong();
091      flags ^= coded_flags;
092    }
093
094    int size = fc.dataSizeLsb;
095
096    int stream_id;
097    long coded_pts;
098    int header_idx = fc.headerIdx;
099    int frame_res = fc.reservedCount;
100
101    if ((flags & FLAG_STREAM_ID) == FLAG_STREAM_ID) {
102      stream_id = in.readVarInt();
103      if (stream_id >= nut.streams.size()) {
104        throw new IOException("Illegal stream id value " + stream_id + " must be < "
105            + nut.streams.size());
106      }
107    } else {
108      stream_id = fc.streamId;
109    }
110
111    stream = nut.streams.get(stream_id);
112
113
114    if ((flags & FLAG_CODED_PTS) == FLAG_CODED_PTS) {
115      coded_pts = in.readVarLong();
116      if (coded_pts < (1 << stream.header.msbPtsShift)) {
117        long mask = (1L << stream.header.msbPtsShift) - 1;
118        long delta = stream.last_pts - mask / 2;
119        pts = ((coded_pts - delta) & mask) + delta;
120      } else
121        pts = coded_pts - (1L << stream.header.msbPtsShift);
122    } else {
123      // TODO Test this code path
124      pts = stream.last_pts + fc.ptsDelta;
125    }
126    stream.last_pts = pts;
127
128    if ((flags & FLAG_SIZE_MSB) == FLAG_SIZE_MSB) {
129      int data_size_msb = in.readVarInt();
130      size += fc.dataSizeMul * data_size_msb;
131    }
132    if ((flags & FLAG_MATCH_TIME) == FLAG_MATCH_TIME) {
133      fc.matchTimeDelta = in.readSignedVarInt();
134    }
135    if ((flags & FLAG_HEADER_IDX) == FLAG_HEADER_IDX) {
136      header_idx = in.readVarInt();
137      if (header_idx >= nut.header.elision.size()) {
138        throw new IOException("Illegal header index " + header_idx + " must be < "
139            + nut.header.elision.size());
140      }
141    }
142    if ((flags & FLAG_RESERVED) == FLAG_RESERVED) {
143      frame_res = in.readVarInt();
144    }
145
146    for (int i = 0; i < frame_res; i++) {
147      in.readVarLong(); // Discard
148    }
149
150    if ((flags & FLAG_CHECKSUM) == FLAG_CHECKSUM) {
151      long checksum = in.readInt();
152      // TODO Test checksum
153    }
154
155    if (size > 4096) {
156      header_idx = 0;
157    }
158
159    // Now data
160    if ((flags & FLAG_SM_DATA) == FLAG_SM_DATA) {
161      // TODO Test this path.
162
163      if (nut.header.version < 4) {
164        throw new IOException("Frame SM Data not allowed in version 4 or less");
165      }
166      long pos = in.offset();
167      sideData = readMetaData(in);
168      metaData = readMetaData(in);
169      size -= (in.offset() - pos);
170
171    } else {
172      sideData = null;
173      metaData = null;
174    }
175
176    // TODO Use some kind of byte pool
177    data = new byte[size];
178
179    byte[] elision = nut.header.elision.get(header_idx);
180    System.arraycopy(elision, 0, data, 0, elision.length);
181    in.readFully(data, elision.length, size - elision.length);
182  }
183
184
185  @Override
186  public String toString() {
187    return MoreObjects.toStringHelper(this).add("id", stream.header.id).add("pts", pts)
188        .add("data", String.format("(%d bytes)", data.length)).toString();
189  }
190}
