001package net.bramp.ffmpeg.nut;
002
003
004import com.google.common.io.CountingInputStream;
005import net.bramp.ffmpeg.io.CRC32InputStream;
006
007import java.io.DataInput;
008import java.io.DataInputStream;
009import java.io.IOException;
010import java.io.InputStream;
011
012import static com.google.common.base.Preconditions.checkNotNull;
013
014/**
015 * A DataInputStream that implements a couple of custom FFmpeg Nut datatypes.
016 */
017public class NutDataInputStream implements DataInput {
018
019  final DataInputStream in;
020  final CRC32InputStream crc;
021  final CountingInputStream count;
022
023  // These are for debugging, remove later
024  long startCrcRange;
025  long endCrcRange;
026
027  public NutDataInputStream(InputStream in) {
028    checkNotNull(in);
029    this.count = new CountingInputStream(in);
030    this.crc = new CRC32InputStream(count);
031    this.in = new DataInputStream(crc);
032  }
033
034  public void resetCRC() {
035    startCrcRange = count.getCount();
036    crc.resetCrc();
037  }
038
039  public long getCRC() {
040    endCrcRange = count.getCount();
041    return crc.getValue();
042  }
043
044  // Read a simple var int up to 32 bits
045  public int readVarInt() throws IOException {
046    boolean more;
047    int result = 0;
048    do {
049      int b = in.readUnsignedByte();
050      more = (b & 0x80) == 0x80;
051      result = 128 * result + (b & 0x7F);
052
053      // TODO Check for int overflow
054    } while (more);
055
056    return result;
057  }
058
059  // Read a simple var int up to 64 bits
060  public long readVarLong() throws IOException {
061    boolean more;
062    long result = 0;
063    do {
064      int b = in.readUnsignedByte();
065      more = (b & 0x80) == 0x80;
066      result = 128 * result + (b & 0x7F);
067
068      // TODO Check for long overflow
069    } while (more);
070
071    return result;
072  }
073
074  // Read a signed var int
075  public long readSignedVarInt() throws IOException {
076    long temp = readVarLong() + 1;
077    if ((temp & 1) == 1) {
078      return -(temp >> 1);
079    }
080    return temp >> 1;
081  }
082
083  // Read a array with a varint prefixed length
084  public byte[] readVarArray() throws IOException {
085    int len = (int) readVarLong();
086    byte[] result = new byte[len];
087    in.read(result);
088    return result;
089  }
090
091  // Returns the start code, OR frame_code if the code doesn't start with 'N'
092  public long readStartCode() throws IOException {
093    byte frameCode = in.readByte();
094    if (frameCode != 'N') {
095      return (long) (frameCode & 0xff);
096    }
097
098    // Otherwise read the remaining 64bit startCode
099    byte[] buffer = new byte[8];
100    buffer[0] = frameCode;
101    readFully(buffer, 1, 7);
102    return (((long) buffer[0] << 56) + ((long) (buffer[1] & 255) << 48)
103        + ((long) (buffer[2] & 255) << 40) + ((long) (buffer[3] & 255) << 32)
104        + ((long) (buffer[4] & 255) << 24) + ((buffer[5] & 255) << 16) + ((buffer[6] & 255) << 8) + ((buffer[7] & 255) << 0));
105  }
106
107  public long offset() {
108    return count.getCount();
109  }
110
111  @Override
112  public void readFully(byte[] b) throws IOException {
113    in.readFully(b);
114  }
115
116  @Override
117  public void readFully(byte[] b, int off, int len) throws IOException {
118    in.readFully(b, off, len);
119  }
120
121  @Override
122  public int skipBytes(int n) throws IOException {
123    return in.skipBytes(n);
124  }
125
126  @Override
127  public boolean readBoolean() throws IOException {
128    return in.readBoolean();
129  }
130
131  @Override
132  public byte readByte() throws IOException {
133    return in.readByte();
134  }
135
136  @Override
137  public int readUnsignedByte() throws IOException {
138    return in.readUnsignedByte();
139  }
140
141  @Override
142  public short readShort() throws IOException {
143    return in.readShort();
144  }
145
146  @Override
147  public int readUnsignedShort() throws IOException {
148    return in.readUnsignedShort();
149  }
150
151  @Override
152  public char readChar() throws IOException {
153    return in.readChar();
154  }
155
156  @Override
157  public int readInt() throws IOException {
158    return in.readInt();
159  }
160
161  @Override
162  public long readLong() throws IOException {
163    return in.readLong();
164  }
165
166  @Override
167  public float readFloat() throws IOException {
168    return in.readFloat();
169  }
170
171  @Override
172  public double readDouble() throws IOException {
173    return in.readDouble();
174  }
175
176  @Override
177  @Deprecated
178  public String readLine() throws IOException {
179    return in.readLine();
180  }
181
182  @Override
183  public String readUTF() throws IOException {
184    return in.readUTF();
185  }
186
187}
