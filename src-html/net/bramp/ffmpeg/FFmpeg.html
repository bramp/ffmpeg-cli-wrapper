063  final static Pattern CODECS_REGEX = Pattern
064      .compile("^ ([ D][ E][VAS][ S][ D][ T]) (\\S+)\\s+(.*)$");
065  final static Pattern FORMATS_REGEX = Pattern.compile("^ ([ D][ E]) (\\S+)\\s+(.*)$");

104  public FFmpeg(@Nonnull String path, @Nonnull ProcessFunction runFunction) throws IOException {
105    Preconditions.checkArgument(!Strings.isNullOrEmpty(path));
106    this.runFunc = checkNotNull(runFunction);
107    this.path = path;
108    version();
109  }
110
111  private static BufferedReader wrapInReader(Process p) {
112    return new BufferedReader(new InputStreamReader(p.getInputStream(), Charsets.UTF_8));
113  }
114
115  public synchronized @Nonnull String version() throws IOException {
116    if (this.version == null) {
117      Process p = runFunc.run(ImmutableList.of(path, "-version"));
118      try {
119        BufferedReader r = wrapInReader(p);
120        this.version = r.readLine();
121        IOUtils.copy(r, new NullOutputStream()); // Throw away rest of the output
122        FFmpegUtils.throwOnError(FFMPEG, p);
123      } finally {
124        p.destroy();
125      }
126    }
127    return version;
128  }
129
130  public synchronized @Nonnull List<Codec> codecs() throws IOException {
131    if (this.codecs == null) {
132      codecs = new ArrayList<Codec>();
133
134      Process p = runFunc.run(ImmutableList.of(path, "-codecs"));
135      try {
136        BufferedReader r = wrapInReader(p);
137        String line;
138        while ((line = r.readLine()) != null) {
139          Matcher m = CODECS_REGEX.matcher(line);
140          if (!m.matches())
141            continue;
142
143          codecs.add(new Codec(m.group(2), m.group(3), m.group(1)));
144        }
145
146        FFmpegUtils.throwOnError(FFMPEG, p);
147        this.codecs = ImmutableList.copyOf(codecs);
148      } finally {
149        p.destroy();
150      }
151    }
152
153    return codecs;
154  }
155
156
157  public synchronized @Nonnull List<Format> formats() throws IOException {
158    if (this.formats == null) {
159      formats = new ArrayList<Format>();
160
161      Process p = runFunc.run(ImmutableList.of(path, "-formats"));
162      try {
163        BufferedReader r = wrapInReader(p);
164        String line;
165        while ((line = r.readLine()) != null) {
166          Matcher m = FORMATS_REGEX.matcher(line);
167          if (!m.matches())
168            continue;
169
170          formats.add(new Format(m.group(2), m.group(3), m.group(1)));
171        }
172
173        FFmpegUtils.throwOnError(FFMPEG, p);
174        this.formats = ImmutableList.copyOf(formats);
175      } finally {
176        p.destroy();
177      }
178    }
179    return formats;
180  }
181
182  public void run(List<String> args) throws IOException {
183    List<String> newArgs = ImmutableList.<String>builder().add(path).addAll(args).build();
184
185    Process p = runFunc.run(newArgs);
186    try {
187      // Now block reading ffmpeg's stdout. We are effectively throwing away the output.
188      IOUtils.copy(wrapInReader(p), System.out); // TODO Should I be outputting to stdout?
189
190      FFmpegUtils.throwOnError(FFMPEG, p);
191
192    } finally {
193      p.destroy();
194    }
195  }

199  }

201  public String getPath() {
202    return path;
203  }
