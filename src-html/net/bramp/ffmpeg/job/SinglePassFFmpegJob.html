001package net.bramp.ffmpeg.job;
002
003import com.google.common.base.Throwables;
004import net.bramp.ffmpeg.FFmpeg;
005
006import java.util.List;
007
008public class SinglePassFFmpegJob extends FFmpegJob {
009
010  final List<String> args;
011
012  public SinglePassFFmpegJob(FFmpeg ffmpeg, List<String> args) {
013    super(ffmpeg);
014    this.args = args;
015  }
016
017  public void run() {
018    state = State.RUNNING;
019
020    try {
021      ffmpeg.run(args);
022      state = State.FINISHED;
023
024    } catch (Throwable t) {
025      state = State.FAILED;
026      Throwables.propagate(t);
027    }
028  }
029}
