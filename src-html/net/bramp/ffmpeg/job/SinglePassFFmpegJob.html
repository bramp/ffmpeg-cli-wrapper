001package net.bramp.ffmpeg.job;
002
003import com.google.common.base.Throwables;
004import net.bramp.ffmpeg.FFmpeg;
005import net.bramp.ffmpeg.builder.FFmpegBuilder;
006import net.bramp.ffmpeg.progress.ProgressListener;
007
008import javax.annotation.Nullable;
009
010import static com.google.common.base.Preconditions.checkNotNull;
011
012public class SinglePassFFmpegJob extends FFmpegJob {
013
014  final public FFmpegBuilder builder;
015
016  public SinglePassFFmpegJob(FFmpeg ffmpeg, FFmpegBuilder builder) {
017    this(ffmpeg, builder, null);
018  }
019
020  public SinglePassFFmpegJob(FFmpeg ffmpeg, FFmpegBuilder builder,
021      @Nullable ProgressListener listener) {
022    super(ffmpeg, listener);
023    this.builder = checkNotNull(builder);
024
025    // Build the args now (but throw away the results). This allows the illegal arguments to be
026    // caught early, but also allows the ffmpeg command to actually alter the arguments when
027    // running.
028    this.builder.build();
029  }
030
031  public void run() {
032
033    state = State.RUNNING;
034
035    try {
036      ffmpeg.run(builder, listener);
037      state = State.FINISHED;
038
039    } catch (Throwable t) {
040      state = State.FAILED;
041
042      Throwables.throwIfUnchecked(t);
043      throw new RuntimeException(t);
044    }
045  }
046}
