001package net.bramp.ffmpeg;
002
003import com.google.common.base.Joiner;
004import com.google.common.base.Preconditions;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.io.IOException;
009import java.util.List;
010
011/**
012 * Simple function that creates a Process with the arguments, and returns a BufferedReader reading
013 * stdout
014 * 
015 * @author bramp
016 *
017 */
018public class RunProcessFunction implements ProcessFunction {
019
020  final static Logger LOG = LoggerFactory.getLogger(RunProcessFunction.class);
021
022  public Process run(List<String> args) throws IOException {
023
024    Preconditions.checkNotNull(args, "Arguments must not be null");
025    Preconditions.checkArgument(!args.isEmpty(), "No arguments specified");
026
027    if (LOG.isInfoEnabled()) {
028      LOG.info("{}", Joiner.on(" ").join(args));
029    }
030
031    ProcessBuilder builder = new ProcessBuilder(args);
032    builder.redirectErrorStream(true);
033    return builder.start();
034  }
035
036}
