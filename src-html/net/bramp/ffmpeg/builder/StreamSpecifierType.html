001package net.bramp.ffmpeg.builder;
002
003public enum StreamSpecifierType {
004  /** Video */
005  Video("v"),
006
007  /** Video streams which are not attached pictures, video thumbnails or cover arts. */
008  PureVideo("V"),
009
010  /** Audio */
011  Audio("a"),
012
013  /** Subtitles */
014  Subtitle("s"),
015
016  /** Data */
017  Data("d"),
018
019  /** Attachment */
020  Attachment("t");
021
022  final String prefix;
023
024  StreamSpecifierType(String prefix) {
025    this.prefix = prefix;
026  }
027
028  @Override
029  public String toString() {
030    return prefix;
031  }
032}
