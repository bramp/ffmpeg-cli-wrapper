356  public EncodingOptions buildOptions() {
357    // TODO When/if modelmapper supports @ConstructorProperties, we map this
358    // object, instead of doing new XXX(...)
359    // https://github.com/jhalterman/modelmapper/issues/44
360    return new EncodingOptions(new MainEncodingOptions(format, startOffset, duration),
361        new AudioEncodingOptions(audio_enabled, audio_codec, audio_channels, audio_sample_rate,
362            audio_bit_depth, audio_bit_rate, audio_quality), new VideoEncodingOptions(
363            video_enabled, video_codec, video_frame_rate, video_width, video_height,
364            video_bit_rate, video_frames, video_filter, video_preset));
365  }

367  protected List<String> build(int pass) {
368    Preconditions.checkState(parent != null, "Can not build without parent being set");
369
370    ImmutableList.Builder<String> args = new ImmutableList.Builder<String>();
371
372    if (targetSize > 0) {
373      checkState(parent.inputs.size() == 1, "Target size does not support multiple inputs");
374
375      String filename = parent.inputs.get(0);
376      FFmpegProbeResult input = parent.inputProbes.get(filename);
377
378      checkState(input != null, "Target size must be used with setInput(FFmpegProbeResult)");
379
380      // TODO factor in start time and/or number of frames
381
382      double durationInSeconds = input.format.duration;
383      long totalBitRate =
384          (long) Math.floor((targetSize * 8) / durationInSeconds) - pass_padding_bitrate;
385
386      // TODO Calculate audioBitRate
387
388      if (video_enabled && video_bit_rate == 0) {
389        // Video (and possibly audio)
390        long audioBitRate = audio_enabled ? audio_bit_rate : 0;
391        video_bit_rate = totalBitRate - audioBitRate;
392      } else if (audio_enabled && audio_bit_rate == 0) {
393        // Just Audio
394        audio_bit_rate = totalBitRate;
395      }
396    }
397
398    if (strict != FFmpegBuilder.Strict.NORMAL) {
399      args.add("-strict").add(strict.toString().toLowerCase());
400    }
401
402    if (!Strings.isNullOrEmpty(format)) {
403      args.add("-f").add(format);
404    }
405
406    if (startOffset != null) {
407      args.add("-ss").add(millisecondsToString(startOffset));
408    }
409
410    if (duration != null) {
411      args.add("-t").add(millisecondsToString(duration));
412    }
413
414    if (video_enabled) {
415
416      if (video_frames != null) {
417        args.add("-vframes").add(String.format("%d", video_frames));
418      }
419
420      if (!Strings.isNullOrEmpty(video_codec)) {
421        args.add("-vcodec").add(video_codec);
422      }
423
424      if (video_width != 0 && video_height != 0) {
425        args.add("-s").add(String.format("%dx%d", video_width, video_height));
426      }
427
428      if (video_frame_rate != null) {
429        // args.add("-r").add(String.format("%2f", video_frame_rate));
430        args.add("-r").add(video_frame_rate.toString());
431      }
432
433      if (video_bit_rate > 0) {
434        args.add("-b:v").add(String.format("%d", video_bit_rate));
435      }
436
437      if (!Strings.isNullOrEmpty(video_preset)) {
438        args.add("-vpre").add(video_preset);
439      }
440
441      if (!Strings.isNullOrEmpty(video_filter)) {
442        checkState(parent.inputs.size() == 1,
443            "Video filter only works with one input, instead use setVideoFilterComplex(..)");
444        args.add("-vf").add(video_filter);
445      }
446
447      if (!Strings.isNullOrEmpty(video_filter_complex)) {
448        args.add("-filter_complex").add(video_filter_complex);
449      }
450
451    } else {
452      args.add("-vn");
453    }
454
455    if (audio_enabled && pass != 1) {
456      if (!Strings.isNullOrEmpty(audio_codec)) {
457        args.add("-acodec").add(audio_codec);
458      }
459
460      if (audio_channels > 0) {
461        args.add("-ac").add(String.format("%d", audio_channels));
462      }
463
464      if (audio_sample_rate > 0) {
465        args.add("-ar").add(String.format("%d", audio_sample_rate));
466      }
467
468      if (!Strings.isNullOrEmpty(audio_bit_depth)) {
469        args.add("-sample_fmt").add(audio_bit_depth);
470      }
471
472      if (audio_bit_rate > 0 && audio_quality > 0 && throwWarnings) {
473        // I'm not sure, but it seems audio_quality overrides
474        // audio_bit_rate, so don't allow both
475        throw new IllegalStateException("Only one of audio_bit_rate and audio_quality can be set");
476      }
477
478      if (audio_bit_rate > 0) {
479        args.add("-b:a").add(String.format("%d", audio_bit_rate));
480      }
481
482      if (audio_quality > 0) {
483        args.add("-aq").add(String.format("%d", audio_quality));
484      }
485
486    } else {
487      args.add("-an");
488    }
489
490    if (!subtitle_enabled)
491      args.add("-sn");
492
493    // Output
494    if (pass == 1) {
495      args.add(DEVNULL);
496    } else {
497      args.add(filename);
498    }
499
500    return args.build();
501  }
