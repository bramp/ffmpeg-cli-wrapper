001package net.bramp.ffmpeg.builder;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004import static net.bramp.ffmpeg.builder.MetadataSpecifier.checkValidKey;
005
006/**
007 *
008 * https://ffmpeg.org/ffmpeg.html#Stream-specifiers
009 */
010public class StreamSpecifier {
011
012  final String spec;
013
014  private StreamSpecifier(String spec) {
015    this.spec = spec;
016  }
017
018  public String spec() {
019    return spec;
020  }
021
022  /**
023   * Matches the stream with this index.
024   * 
025   * @param index The stream index
026   * @return A new StreamSpecifier
027   */
028  public static StreamSpecifier stream(int index) {
029    return new StreamSpecifier(String.valueOf(index));
030  }
031
032  /**
033   * Matches all streams of this type.
034   * 
035   * @param type The stream type
036   * @return A new StreamSpecifier
037   */
038  public static StreamSpecifier stream(StreamSpecifierType type) {
039    checkNotNull(type);
040    return new StreamSpecifier(type.toString());
041  }
042
043  /**
044   * Matches the stream number stream_index of this type.
045   * 
046   * @param type The stream type
047   * @param index The stream index
048   * @return A new StreamSpecifier
049   */
050  public static StreamSpecifier stream(StreamSpecifierType type, int index) {
051    checkNotNull(type);
052    return new StreamSpecifier(type.toString() + ":" + index);
053  }
054
055  /**
056   * Matches all streams in the program.
057   * 
058   * @param program_id The program id
059   * @return A new StreamSpecifier
060   */
061  public static StreamSpecifier program(int program_id) {
062    return new StreamSpecifier("p:" + program_id);
063  }
064
065  /**
066   * Matches the stream with number stream_index in the program with the id program_id.
067   * 
068   * @param program_id The program id
069   * @param stream_index The stream index
070   * @return A new StreamSpecifier
071   */
072  public static StreamSpecifier program(int program_id, int stream_index) {
073    return new StreamSpecifier("p:" + program_id + ":" + stream_index);
074  }
075
076  /**
077   * Match the stream by stream id (e.g. PID in MPEG-TS container).
078   *
079   * @param stream_id The stream id
080   * @return A new StreamSpecifier
081   */
082  public static StreamSpecifier id(int stream_id) {
083    return new StreamSpecifier("i:" + stream_id);
084  }
085
086  /**
087   * Matches all streams with the given metadata tag.
088   * 
089   * @param key The metadata tag
090   * @return A new StreamSpecifier
091   */
092  public static StreamSpecifier tag(String key) {
093    return new StreamSpecifier("m:" + checkValidKey(key));
094  }
095
096  /**
097   * Matches streams with the metadata tag key having the specified value.
098   * 
099   * @param key The metadata tag
100   * @param value The metatdata's value
101   * @return A new StreamSpecifier
102   */
103  public static StreamSpecifier tag(String key, String value) {
104    checkValidKey(key);
105    checkNotNull(value);
106    return new StreamSpecifier("m:" + key + ":" + value);
107  }
108
109  /**
110   * Matches streams with usable configuration, the codec must be defined and the essential
111   * information such as video dimension or audio sample rate must be present.
112   * 
113   * @return A new StreamSpecifier
114   */
115  public static StreamSpecifier usable() {
116    return new StreamSpecifier("u");
117  }
118}
