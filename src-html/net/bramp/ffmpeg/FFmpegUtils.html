001package net.bramp.ffmpeg;
002
003import com.google.gson.Gson;
004import com.google.gson.GsonBuilder;
005import net.bramp.commons.lang3.math.gson.FractionAdapter;
006import net.bramp.ffmpeg.gson.NamedBitsetAdapter;
007import net.bramp.ffmpeg.io.ProcessUtils;
008import net.bramp.ffmpeg.probe.FFmpegDisposition;
009import org.apache.commons.lang3.math.Fraction;
010
011import java.io.IOException;
012import java.util.concurrent.TimeUnit;
013import java.util.concurrent.TimeoutException;
014
015import static com.google.common.base.Preconditions.checkArgument;
016
017/**
018 * Helper class with commonly used methods
019 */
020public final class FFmpegUtils {
021
022  final static Gson gson = FFmpegUtils.setupGson();
023
024  private FFmpegUtils() {}
025
026  /**
027   * Convert milliseconds to "hh:mm:ss.ms" String representation.
028   *
029   * @param milliseconds
030   * @return
031   */
032  public static String millisecondsToString(long milliseconds) {
033
034    checkArgument(milliseconds >= 0, "milliseconds must be >= 0");
035
036    long seconds = milliseconds / 1000;
037    milliseconds = milliseconds - (seconds * 1000);
038
039    long minutes = seconds / 60;
040    seconds = seconds - (minutes * 60);
041
042    long hours = minutes / 60;
043    minutes = minutes - (hours * 60);
044
045    if (milliseconds == 0)
046      return String.format("%02d:%02d:%02d", hours, minutes, seconds);
047
048    return String.format("%02d:%02d:%02d.%03d", hours, minutes, seconds, milliseconds);
049  }
050
051  public static void throwOnError(String cmd, Process p) throws IOException {
052    try {
053      // TODO In java 8 use waitFor(long timeout, TimeUnit unit)
054      if (ProcessUtils.waitForWithTimeout(p, 1, TimeUnit.SECONDS) != 0) {
055        // TODO Parse the error
056        throw new IOException(cmd + " returned non-zero exit status. Check stdout.");
057      }
058    } catch (TimeoutException e) {
059      throw new IOException("Timed out waiting for " + cmd + " to finish.");
060    }
061  }
062
063  static Gson getGson() {
064    return gson;
065  }
066
067  private static Gson setupGson() {
068    GsonBuilder builder = new GsonBuilder();
069    builder.registerTypeAdapter(Fraction.class, new FractionAdapter());
070    builder.registerTypeAdapter(FFmpegDisposition.class, new NamedBitsetAdapter<>(
071        FFmpegDisposition.class));
072
073    return builder.create();
074  }
075}
