001package net.bramp.ffmpeg.modelmapper;
002
003import com.google.common.base.Defaults;
004import com.google.common.base.Objects;
005import org.modelmapper.Condition;
006import org.modelmapper.spi.MappingContext;
007
008/**
009 * Only maps properties which are not their type's default value.
010 *
011 * @param <S> source type
012 * @param <D> destination type
013 *
014 * @author bramp
015 */
016public class NotDefaultCondition<S, D> implements Condition<S, D> {
017
018  public final static NotDefaultCondition notDefault = new NotDefaultCondition();
019
020  @Override
021  public boolean applies(MappingContext<S, D> context) {
022    return !Objects.equal(context.getSource(), Defaults.defaultValue(context.getSourceType()));
023  }
024}
