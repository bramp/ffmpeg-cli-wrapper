001package net.bramp.ffmpeg.io;
002
003import java.util.concurrent.TimeUnit;
004import java.util.concurrent.TimeoutException;
005
006/**
007 * @author bramp
008 */
009public abstract class ProcessUtils {
010  private ProcessUtils() {}
011
012  private static class ProcessThread extends Thread {
013    final Process p;
014    boolean finished = false;
015    int exitValue = -1;
016
017    private ProcessThread(Process p) {
018      this.p = p;
019    }
020
021    @Override
022    public void run() {
023      try {
024        exitValue = p.waitFor();
025        finished = true;
026      } catch (InterruptedException e) {
027        // Ignore
028      }
029    }
030
031    public boolean hasFinished() {
032      return finished;
033    }
034
035    public int exitValue() {
036      return exitValue;
037    }
038  }
039
040  /**
041   * Waits until the process finishs or the timeout occurs
042   * 
043   * @param p
044   * @param timeout
045   * @param unit
046   * @return the process' exit value
047   * @throws TimeoutException if the timeout occurs
048   */
049  public static int waitForWithTimeout(final Process p, long timeout, TimeUnit unit)
050      throws TimeoutException {
051
052    ProcessThread t = new ProcessThread(p);
053    t.start();
054    try {
055      unit.timedJoin(t, timeout);
056
057    } catch (InterruptedException e) {
058      t.interrupt();
059    }
060
061    if (!t.hasFinished()) {
062      throw new TimeoutException("Process did not finish within timeout");
063    }
064
065    return t.exitValue();
066  }
067}
