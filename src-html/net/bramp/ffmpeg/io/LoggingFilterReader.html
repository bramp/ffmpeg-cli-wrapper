001package net.bramp.ffmpeg.io;
002
003import com.google.common.primitives.Chars;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import java.io.FilterReader;
008import java.io.IOException;
009import java.io.Reader;
010
011/**
012 * Wraps a Reader, and logs full lines of input as its read.
013 *
014 * @author bramp
015 */
016public class LoggingFilterReader extends FilterReader {
017
018  final static char LOG_CHAR = '\n';
019
020  final Logger logger;
021  final StringBuilder buffer = new StringBuilder();
022
023  public LoggingFilterReader(Reader in, Logger logger) {
024    super(in);
025    this.logger = logger;
026  }
027
028  protected void log() {
029    if (buffer.length() > 0) {
030      // TODO Change from debug, to a user defined level
031      logger.debug(buffer.toString());
032      buffer.setLength(0);
033    }
034  }
035
036  private static int indexOf(char[] array, char c, int off, int len) {
037    for (int i = off; i < off + len; i++) {
038      if (array[i] == c) {
039        return i;
040      }
041    }
042    return -1;
043  }
044
045  @Override
046  public int read(char[] cbuf, int off, int len) throws IOException {
047    int ret = super.read(cbuf, off, len);
048    if (ret != -1) {
049      buffer.append(cbuf, off, ret);
050    }
051
052    // If end of stream, or contains new line
053    if (ret == -1 || indexOf(cbuf, LOG_CHAR, off, ret) != -1) {
054      // BUG this will log a unfinished line, if a string such as
055      // "line \n unfinished" is read.
056      log();
057    }
058
059    return ret;
060  }
061
062  @Override
063  public int read() throws IOException {
064    int ret = super.read();
065    if (ret != -1) {
066      buffer.append((char) ret);
067    }
068
069    // If end of stream, or contains new line
070    if (ret == -1 || ret == LOG_CHAR) {
071      log();
072    }
073    return ret;
074  }
075}
