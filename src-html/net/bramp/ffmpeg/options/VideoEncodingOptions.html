001package net.bramp.ffmpeg.options;
002
003import org.apache.commons.lang3.math.Fraction;
004
005import java.beans.ConstructorProperties;
006
007/**
008 * Encoding options for video
009 * 
010 * @author bramp
011 *
012 */
013public class VideoEncodingOptions {
014  public final boolean enabled;
015  public final String codec;
016  public final Fraction frame_rate;
017  public final int width;
018  public final int height;
019  public final long bit_rate;
020  public final Integer frames;
021  public final String filter;
022  public final String preset;
023
024  @ConstructorProperties({"enabled", "codec", "frame_rate", "width", "height", "bit_rate",
025      "frames", "video_filter", "preset"})
026  public VideoEncodingOptions(boolean enabled, String codec, Fraction frame_rate, int width,
027      int height, long bit_rate, Integer frames, String filter, String preset) {
028    this.enabled = enabled;
029    this.codec = codec;
030    this.frame_rate = frame_rate;
031    this.width = width;
032    this.height = height;
033    this.bit_rate = bit_rate;
034    this.frames = frames;
035    this.filter = filter;
036    this.preset = preset;
037  }
038
039}
