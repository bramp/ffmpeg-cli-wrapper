001package net.bramp.ffmpeg.gson;
002
003import com.google.common.base.Optional;
004import com.google.gson.TypeAdapter;
005import com.google.gson.stream.JsonReader;
006import com.google.gson.stream.JsonToken;
007import com.google.gson.stream.JsonWriter;
008
009import java.io.IOException;
010import java.lang.reflect.Field;
011
012import static com.google.common.base.Preconditions.checkNotNull;
013
014/**
015 * Converts a json object which represents a set of booleans. For example:
016 * 
017 * <pre>
018 * <code>
019 * public class Set {
020 *   public boolean a = true;
021 *   public boolean b = false;
022 *   public int c = 1;
023 *   public int d = 0;
024 * }
025 * </code>
026 * </pre>
027 * 
028 * is turned into:
029 * 
030 * <pre>
031 * {
032 *   "a": true,
033 *   "b": false,
034 *   "c": true,
035 *   "d": false
036 * }
037 * </pre>
038 *
039 */
040public class NamedBitsetAdapter<T> extends TypeAdapter<T> {
041
042  final Class<T> clazz;
043
044  public NamedBitsetAdapter(Class<T> clazz) {
045    this.clazz = checkNotNull(clazz);
046  }
047
048  protected Optional<Boolean> readBoolean(JsonReader reader) throws IOException {
049    JsonToken next = reader.peek();
050    switch (next) {
051      case BOOLEAN:
052        return Optional.of(reader.nextBoolean());
053      case NUMBER:
054        return Optional.of(reader.nextInt() != 0);
055    }
056
057    reader.skipValue();
058
059    return Optional.absent();
060  }
061
062  protected void setField(T target, String name, boolean value) throws IllegalAccessException {
063    try {
064      Field f = clazz.getField(name);
065      if ((boolean.class.equals(f.getType()))) {
066        f.setBoolean(target, value);
067      } else if (int.class.equals(f.getType())) {
068        f.setInt(target, value ? 1 : 0);
069      }
070
071    } catch (NoSuchFieldException e) {
072      // Just continue
073    }
074  }
075
076  public T read(JsonReader reader) throws IOException {
077
078    JsonToken next = reader.peek();
079
080    if (next == JsonToken.NULL) {
081      reader.nextNull();
082      return null;
083    }
084
085    try {
086      T obj = clazz.newInstance();
087      reader.beginObject();
088
089      next = reader.peek();
090      while (next != JsonToken.END_OBJECT) {
091        String name = reader.nextName();
092        Optional<Boolean> value = readBoolean(reader);
093
094        if (value.isPresent()) {
095          setField(obj, name, value.get());
096        }
097
098        next = reader.peek();
099      }
100
101      reader.endObject();
102      return obj;
103
104    } catch (InstantiationException | IllegalAccessException e) {
105      throw new IOException("Reflection error", e);
106    }
107  }
108
109  public void write(JsonWriter writer, T value) throws IOException {
110
111    if (value == null) {
112      writer.nullValue();
113      return;
114    }
115
116    assert value.getClass().equals(clazz);
117
118    writer.beginObject();
119    for (Field f : clazz.getFields()) {
120      try {
121        boolean b;
122        if (boolean.class.equals(f.getType())) {
123          b = f.getBoolean(value);
124        } else if (int.class.equals(f.getType())) {
125          b = f.getInt(value) != 0;
126        } else {
127          continue;
128        }
129
130        writer.name(f.getName());
131        writer.value(b);
132
133      } catch (IllegalAccessException e) {
134        throw new IOException("Reflection error", e);
135      }
136    }
137    writer.endObject();
138  }
139}
