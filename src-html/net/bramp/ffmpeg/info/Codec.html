001package net.bramp.ffmpeg.info;
002
003import com.google.common.base.Preconditions;
004import org.apache.commons.lang3.builder.EqualsBuilder;
005import org.apache.commons.lang3.builder.HashCodeBuilder;
006
007/**
008 * Information about supported Codecs
009 * 
010 * @author bramp
011 *
012 */
013public class Codec {
014
015  enum Type {
016    VIDEO, AUDIO, SUBTITLE
017  }
018
019
020  final String name;
021  final String longName;
022
023  /**
024   * Can I decode with this codec
025   */
026  final boolean canDecode;
027
028  /**
029   * Can I encode with this codec
030   */
031  final boolean canEncode;
032
033  /**
034   * What type of codec is this
035   */
036  final Type type;
037
038  public Codec(String name, String longName, String flags) {
039    this.name = Preconditions.checkNotNull(name).trim();
040    this.longName = Preconditions.checkNotNull(longName).trim();
041
042    /**
043     * {@literal
044     * D..... = Decoding supported
045     * .E.... = Encoding supported
046     * ..V... = Video codec
047     * ..A... = Audio codec
048     * ..S... = Subtitle codec
049     * ...S.. = Supports draw_horiz_band
050     * ....D. = Supports direct rendering method 1
051     * .....T = Supports weird frame truncation
052                 * }
053     */
054
055    Preconditions.checkNotNull(flags);
056    Preconditions.checkArgument(flags.length() == 6, "Format flags is invalid '{}'", flags);
057    this.canDecode = flags.charAt(0) == 'D';
058    this.canEncode = flags.charAt(1) == 'E';
059
060    switch (flags.charAt(2)) {
061      case 'V':
062        this.type = Type.VIDEO;
063        break;
064      case 'A':
065        this.type = Type.AUDIO;
066        break;
067      case 'S':
068        this.type = Type.SUBTITLE;
069        break;
070      default:
071        throw new IllegalArgumentException("Invalid codec type '" + flags.charAt(3) + "'");
072    }
073
074    // TODO There are more flags to parse
075  }
076
077  @Override
078  public String toString() {
079    return name + " " + longName;
080  }
081
082  @Override
083  public boolean equals(Object obj) {
084    return EqualsBuilder.reflectionEquals(this, obj);
085  }
086
087  @Override
088  public int hashCode() {
089    return HashCodeBuilder.reflectionHashCode(this);
090  }
091
092  public String getName() {
093    return name;
094  }
095
096  public String getLongName() {
097    return longName;
098  }
099
100  public boolean getCanDecode() {
101    return canDecode;
102  }
103
104  public boolean getCanEncode() {
105    return canEncode;
106  }
107
108  public Type getType() {
109    return type;
110  }
111
112}
